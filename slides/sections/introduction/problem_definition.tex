\begin{frame}
\frametitle{Longest Common Subsequence}
\framesubtitle{Problem Definition}

\textbf{Input:}

Two strings $x$, $y$ over the alphabet $\Sigma$.

\textbf{Output:}

Longest string $z$ such that there exist indices $i_k, j_k$ with $i_k \leq j_k < i_{k+1}$ \only<3->{and $i'_k, j'_k$ with $i'_k \leq j'_k < i'_{k+1}$} such that:

\begin{itemize}
	\item<2-> $z = x[i_{1}..j_{1}] \cdot x[i_{2}..j_{2}] \cdots x[i_{K}..j_{K}]$
	\item<3-> $z = y[i'_{1}..j'_{1}] \cdot y[i'_{2}..j'_{2}] \cdots y[i'_{K'}..j'_{K'}]$
\end{itemize}
