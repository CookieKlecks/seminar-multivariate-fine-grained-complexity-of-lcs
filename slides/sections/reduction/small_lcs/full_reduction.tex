	\end{enumerate}

\visible<1->{
\begin{theorem}
	Let $\mathbf{\alpha}$ be a parameter setting with \emphTheorem{$\alpha_\Sigma = 0$} and \emphTheorem{$\alpha_\delta = \alpha_m$}. 
	There is $\gamma \geq 1$ such that $\lcsy{\mathbf{\alpha}}$ takes time $d^{1-o(1)}$, unless OVH fails.
\end{theorem}	
}	

\only<2>{
\begin{definition}[Parameter Setting (recall)]
		\begin{itemize}
			\setlength{\itemindent}{1em}
			\item Set of parameters $\mathcal{P}$,\quad $\mathbf{\alpha} := \alpha_{p \in \mathcal{P}}$
			\item $\lcsy{\mathbf{\alpha}}$ defined as problem of computing LCS of $x$ and $y$, where:
			\begin{displaymath}
				\frac{n^{\alpha_p}}{\gamma} \leq p(x,y) \leq n^{\alpha_p}\gamma, \quad \text{for all } p \in \mathcal{P}
			\end{displaymath}
		\end{itemize}
\end{definition}
}

\visible<3->{
\begin{proof}
\textbf{Given:} parameter setting $\mathbf{\alpha}$ and $n \geq 1$. \textbf{Notation:} $p = n^{\alpha_p}$ for $p \in \mathcal{P}$ as \emphBeamer{target value} of this parameter.
\vspace{-1ex}
\begin{columns}
	\column[t]{0.01\textwidth}
	\column[t]{0.99\textwidth}
	\begin{itemize}
		\item<4-> use construction with \emphTheorem{$A := \lfloor L/D \rfloor$} and \emphTheorem{$B := \lfloor d/(LD) \rfloor$}
		\item<5-> UOVH implies lower bound for running time: $(AB)^{1 - o(1)} = (d/D^2)^{1 - o(1)} = n^{\alpha_d - o(1)} = d^{1 - o(1)}$.
		\item<6-> reduction time $\bigO{AD} = \bigO{L} \leq \bigO{d^{1-\varepsilon}}$
		\item<7-> seen before: $L(x,y) \geq \rho \Leftrightarrow \exists a_i, b_j:\; \langle a_i,b_j \rangle = 0$ \quad\quad (for some appropriate $\rho$)
	\end{itemize}		
	
\end{columns}
\end{proof}

}
