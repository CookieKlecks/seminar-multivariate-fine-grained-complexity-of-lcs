	\begin{definition}[Guarding of Input Strings]
		For strings $x_1, ..., x_n$ and $y_1, ..., y_m$ with $n \geq m$, we define:
		\begin{columns}[onlytextwidth]
	    \column[t]{0.7\hsize}
	    \vspace{-8pt}
		\begin{align*}
			\guardX{x_1, ..., x_n} &:=& 				  &G(x_1)\; 0^{\gamma_3}\; G(x_2)\; 0^{\gamma_3}\; \cdots\; G(x_{n-1})\; 0^{\gamma_3}\; G(x_n) \\
			\guardY{y_1, ..., y_m} &:=& 0^{n\gamma_4}\; &G(y_1)\; 0^{\gamma_3}\; G(y_2)\; 0^{\gamma_3}\; \cdots\; G(y_{m-1})\; 0^{\gamma_3}\; G(y_m)\; 0^{n\gamma_4}
		\end{align*}

	    \column[t]{0.29\hsize}
	    \vspace{3pt}
		\begin{displaymath}
			G(x) = 1^{\gamma_2}0^{\gamma_1}x0^{\gamma_1}1^{\gamma_2}
		\end{displaymath}
		\end{columns}
		
		with sufficiently large $\gamma_1, \gamma_2, \gamma_3, \gamma_4$
	
	\end{definition}
