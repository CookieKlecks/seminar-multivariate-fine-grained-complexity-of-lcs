	\only<3->{
	\begin{theorem}
		For any two vectors $a_i, b_j \in \{0,1\}^D$, we can construct \emphBeamer{vector gadgets} $\vg{a_i}$ and $\vg{b_j}$, such that:
		\begin{align*}
			\text{if } \langle a_i, b_j \rangle = 0&:&			L(\vg{a_i}, \vg{b_j}) = \rho_0\\
			\text{otherwise}&:&									L(\vg{a_i}, \vg{b_j}) < \rho_0
		\end{align*}				
		
		and $|\vg{a_i}| \leq \bigO{D}$.
	\end{theorem}
