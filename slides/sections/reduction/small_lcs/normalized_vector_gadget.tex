	\begin{theorem}
	For any two vectors $a_i$, $b_j$ and appropriate $\rho_0 > \rho_1$:
	\[
	L(\nvg{a_i}, \nvg{b_j}) = \begin{cases}
		(d+1)\rho_0 & if \langle a_i, b_j \rangle = 0\\
		d\rho_0 + \rho_1 & otherwise
	\end{cases}
	\]
	\end{theorem}
