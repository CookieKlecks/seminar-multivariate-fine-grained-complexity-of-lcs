\visible<2->{
\begin{theorem}
	Let $\mathbf{\alpha}$ be a parameter setting with \emphTheorem{$\alpha_\Sigma > 0$} and \emphTheorem{$\alpha_\delta = \alpha_m$}. 
	There is $\gamma \geq 1$ such that $\lcsy{\mathbf{\alpha}}$ takes time $d^{1-o(1)}$, unless OVH fails.
\end{theorem}	
}	


\visible<3->{
\begin{proof}
\textbf{Given:} parameter setting $\mathbf{\alpha}$ and $n \geq 1$.
\vspace{-1ex}
\begin{columns}
	\column[t]{0.01\textwidth}
	\column[t]{0.99\textwidth}
	\begin{itemize}
		\item<4-> use construction with \emphTheorem{$A := \left\lfloor \frac{d}{\min\{L, \sqrt{d}\} \cdot D} \right\rfloor$} and \emphTheorem{$B := \left\lfloor \frac{\min\{L, \sqrt{d}\}}{D} \right\rfloor$}
		\item<5-> UOVH implies lower bound for running time: $(AB)^{1 - o(1)} = (d/D^2)^{1 - o(1)} = n^{\alpha_d - o(1)} = d^{1 - o(1)}$.
		\item<6-> reduction time $\bigO{AD} \leq \bigO{d^{1-\varepsilon}}$
		\item<7-> seen before: $L(x_i,y_i) \geq \rho \Leftrightarrow \exists a_i' \in A_i, b_j \in B: \langle a_i',b_j \rangle = 0$
	\end{itemize}		
\end{columns}
\end{proof}

}
