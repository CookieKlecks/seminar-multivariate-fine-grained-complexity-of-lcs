	\begin{definition}[Coordinate Gadget]
		For an OV-instance $\mathcal{A} = \{a_1, ..., a_A\}$ and $\mathcal{B} = \{b_1, ..., b_B\}$, we define:
		\begin{columns}[onlytextwidth]
	    \column[t]{0.49\hsize}\parskip\medskipamount
		\[
			\cg{a_i}{k} :=
			\begin{cases}
			\cgZeroX & \text{if } a_i[k] = 0 \\
			\cgOneX & \text{if } a_i[k] = 1
			\end{cases}
			\quad 1 \leq k \leq D,
		\]

	    \column[t]{0.49\hsize}\parskip\medskipamount
		\[
			\text{CG}(b_j, k) :=
			\begin{cases}
			\cgZeroY & \text{if } b_j[k] = 0 \\
			\cgOneY & \text{if } b_j[k] = 1
			\end{cases}
			\quad 1 \leq k \leq D,
		\]
		\end{columns}
	\end{definition}
