\begin{theorem}
	Let $\mathbf{\alpha}$ be a parameter setting with \emphTheorem{$\alpha_\Sigma = 0$} and \emphTheorem{$\alpha_L = \alpha_m$}. 
	There is $\gamma \geq 1$ such that $\lcsy{\mathbf{\alpha}}$ takes time $\min\{d, \delta m, \delta \Delta\}^{1 - o(1)}$, unless OVH fails.
\end{theorem}

\visible<2->{
\begin{proof}
\textbf{Given:} parameter setting $\mathbf{\alpha}$ and $n \geq 1$. 
\vspace{-1ex}
\begin{columns}
	\column[t]{0.01\textwidth}
	\column[t]{0.99\textwidth}
	\begin{itemize}
		\item<2-> use construction with \emphTheorem{$A := \left\lfloor \frac{1}{D} \min\left\{ \delta, \frac{d}{\min\{m, \Delta\}} \right\} \right\rfloor$} and \emphTheorem{$B := \left\lfloor \frac{1}{D} \min\{m, \Delta\} \right\rfloor$}
		\item<3-> UOVH implies lower bound for running time: $(AB)^{1 - o(1)} = \min\{d, \delta m, \delta \Delta\}^{1 - o(1)}$.
		\item<5-> To show: Unique alignment of $x_i$ and $y_j$ $\Rightarrow L(x,y) \geq \rho \Leftrightarrow \exists a_i, b_j:\; \langle a_i,b_j \rangle = 0$
	\end{itemize}		
	
\end{columns}
\end{proof}
