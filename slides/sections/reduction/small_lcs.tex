\begin{frame}[t]
	\frametitle{Reduction}
	
	\begin{definition}[Orthogonal Vectors Problem (OV)]
		\begin{description}
			\item[Given:] Sets $A, B \subset \{0,1\}^D$ \\
				with $|A| = n$, $|B| = m$ 
			
			\item[Decide:] Exists $a \in A$, $b \in B$ with $\langle a,b\rangle = 0$
		\end{description}				
	\end{definition}
	
	\begin{definition}[Unbalanced Orthogonal Vectors Hypothesis (UOVH)]
		The following problem requires time $(n  m)^{1-o(1)}$:
		
		Given a \textbf{number} $\mathbf{N}$, solve an OV instance with:	
\begin{align*} 
	D &= N^{o(1)} \\ 
	|A| &= n \leq N,\quad |B| = m \leq N
\end{align*}
  	\end{definition}
