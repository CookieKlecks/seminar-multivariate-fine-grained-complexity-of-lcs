\newcommand{\sLCSnvg}[3]{%
	\node[base, number, input, minimum width=4.5em, #3] (#1) at (base) {$\nvg{#2}$};
}
\newcommand{\sLCSnvgWithGuard}[4]{%
	\sLCSnvg{#1}{#2}{#4}
	\node[base, number, guard] (#3) at ($(#1.east)$) {$0^\gamma$};
}
